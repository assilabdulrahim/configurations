Phase 3: The Mapping Script Architecture
<#
.SYNOPSIS
    Maps GN100 Blackwell Samba Shares to local Windows Drive Letters.
    Bypasses Active Directory Domain Controller lookups.
#>

$ServerIP = "gn100-3692"
$Shares = @{
    "S" = "\\$ServerIP\FileShare"
    "K" = "\\$ServerIP\KnowledgeBase"
}

Write-Output "Initializing Safe SMB Mapping Protocol..."

# Step 1: Force clean existing stale or disconnected sessions to avoid drive locks
foreach ($Drive in $Shares.Keys) {
    if (Get-PSDrive -Name $Drive -ErrorAction SilentlyContinue) {
        Remove-PSDrive -Name $Drive -Force -ErrorAction SilentlyContinue
    }
    # Clear legacy Windows network routing cache natively
    net use "${Drive}:" /delete /y 2>$null
}

# Step 2: Mount shares explicitly using local-node context
foreach ($Drive in $Shares.Keys) {
    $RemotePath = $Shares[$Drive]
    try {
        Write-Output "Mounting ${RemotePath} to ${Drive}:..."
        
        # Explicit user flag inherits the cached credentials from Phase 2 seamlessly
        net use "${Drive}:" $RemotePath /user:gn100-3692\assilabdulrahim /persistent:yes
        
        Write-Output "Successfully mapped ${Drive}:"
    }
    catch {
        Write-Warning "Failed to map ${Drive}: to ${RemotePath}."
    }
}
